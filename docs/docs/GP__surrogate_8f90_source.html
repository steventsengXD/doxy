669  subroutine svd_inverse(data_pres, mat_size)
670  implicit none
671  integer, intent(in) :: mat_size
672  real(dp), dimension(mat_size,mat_size), intent(inout) :: data_pres
673  real(dp), dimension(mat_size,mat_size) :: U,VT,S_inv, data_temp
674  real(dp), dimension(mat_size) :: S
675  integer, dimension(8*mat_size) :: iwork
676  real(dp), dimension(:,:), allocatable :: work
677  real(dp) :: max_s, min_neg_eig
678  logical :: bad_eigvals, not_psd
679  integer :: info, Lwork, i
680  min_neg_eig=0.0_dp
681  bad_eigvals = .false.
682  not_psd = .false.
683  allocate(work(1,1))
684  call dgesdd('A',mat_size,mat_size,data_temp,mat_size,s,u,mat_size,vt,mat_size,work,-1,iwork,info)
685  if (info.ne.0) then
686  print*, 'Lapack error (SVD query, dgesdd), info=',info
687  stop
688  end if
689  data_temp=data_pres !resetting
690  lwork = int(work(1,1))
691  deallocate(work)
692  allocate(work(lwork,lwork))
693  call dgesdd('A',mat_size,mat_size,data_temp,mat_size,s,u,mat_size,vt,mat_size,work,lwork,iwork,info)
694  if (info.ne.0) then
695  print*, 'Lapack error (SVD, dgesdd), info=',info
696  stop
697  end if
698  max_s=maxval(s)
699  s_inv = 0
700  do i=1,mat_size
701  if (abs(s(i))<epsilon(max_s)*max_s*(mat_size)) then
702  s_inv(i,i)=0
703  bad_eigvals = .true.
704  else if (s(i) <0) then
705  not_psd = .true.
706  s_inv(i,i) = 0!1/s(i)
707  min_neg_eig = min(s(i),min_neg_eig)
708  else
709  s_inv(i,i) = 1/s(i)
710  end if
711  end do
712  if (bad_eigvals) print*, 'Warning, Covariance matrix close to singular'
713  if (not_psd) then
714  print*,'Matrix not PSD, with min eigenvalue:', min_neg_eig
715  end if
716  data_pres = matmul(transpose(vt),matmul(s_inv,transpose(u)))
717 
718  deallocate(work)
719 
720  end subroutine svd_inverse
