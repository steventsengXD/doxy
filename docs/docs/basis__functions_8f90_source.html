302  function discrete_laplacian_reduced(position, h, dofs)
303  ! Computes finite difference approximation to the reduced laplacian $(\nabla^2\psi)/\psi$.
304  ! Uses that $f''(x)/f(x) =((f(x+h)+f(x-h))/f(x)-2)/(h^2)+O(h^2)$ (*)
305  implicit none
306  real(dp) :: discrete_laplacian_reduced
307  real(dp), dimension(:), intent(in) :: position ! Space coordinate of electrons
308  real(dp), intent(in) :: h ! Finite difference lengthscale
309  real(dp), dimension(:), intent(in) :: dofs ! Values of the dofs
310  integer :: i ! Loop variable
311  real(dp), dimension(:,:), allocatable :: delta ! temporary array for vector displacements
312 
313  ! Initialise and bounds tests
314  if (.not.(initialised)) then
315  print *, "Error, basis not initialised"
316  stop
317  end if
318  if (size(position).ne.n_space_dims) then
319  print *, "Error, wrong space dimension, got", size(position), "wanted", n_space_dims
320  stop
321  end if
322  if (size(dofs).ne.number_dofs) then
323  print *, "Error, wrong number of dofs, got", size(dofs), "wanted", number_dofs
324  stop
325  end if
326 
327  ! Create matrix of cartesian basis vectors of length h. Set as 0 initially, h added during main loop
328  allocate(delta(n_space_dims,n_space_dims))
329  delta = 0.0_dp ! Matrix assignment
330  discrete_laplacian_reduced = 0.0_dp
331  do i = 1, n_space_dims ! Loop over space dimensions
332  delta(i,i) = h
333  discrete_laplacian_reduced = discrete_laplacian_reduced &
334  ! Following (*), sum of (f(x+h)+f(x-h)) in all space dimensions
335  + wave_function(position + delta(:,i), dofs) + wave_function(position - delta(:,i), dofs)
336  end do
337  ! Following (*), divide by $f(x)$, and subtract 2 for each space dimension
338  discrete_laplacian_reduced = discrete_laplacian_reduced/wave_function(position ,dofs) - 2*n_space_dims
339  ! Following (*), divide by h^2
340  discrete_laplacian_reduced = discrete_laplacian_reduced/(h**2)
341 
342  ! deallocate temporary array
343  deallocate(delta)
344  end function discrete_laplacian_reduced
