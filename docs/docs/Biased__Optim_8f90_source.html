24  function gaussian(density_dimension) result(x_prop)
25  implicit none
26  integer, intent(in) :: density_dimension
27  real(dp), dimension(2*((density_dimension+1)/2)) :: u !uniform random sample,
28  !note the dimension is the nearest even number above the given dimension
29  !integer division will autmatically round towards 0
30  real(dp), dimension(density_dimension) :: xi !normal random sample
31  real(dp), dimension(density_dimension) :: x_prop
32  integer :: i
33  !proposes, uses same dist as described in log_prop_distinteger, dimension(:), allocatable:: seed
34  !if proposal is changed, might need updated
35 
36  !implementing box muller to generate a normal
37  call random_number(u)
38  do i=1,density_dimension/2
39  xi(i) = sqrt(-2.0_dp*log(u(i)))*cos(2.0_dp*pi*u(i*2))
40  xi(i*2) = sqrt(-2.0_dp*log(u(i)))*sin(2.0_dp*pi*u(i*2))
41  end do
42  if (modulo(density_dimension,2).ne.0) then
43  xi(density_dimension) = sqrt(-2.0_dp*log(u(density_dimension)))*cos(2.0_dp*pi*u(density_dimension+1))
44  end if
45 
46  !basic GRW
47  x_prop = xi
48 
49  end function gaussian
