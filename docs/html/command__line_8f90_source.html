76  ! This function can be called multiple times
77  ! and will freshly parse ALL arguments each time
78  ! We assume these are entered as 'name=value' and
79  ! if there is no '=' sign, then value is set to a sentinel
80  ! For most flexibility, we assume all values are reals, and

86  ! Strictly we can't be sure 50 chars is enough
87  ! but for command-line args it's enough if we're sensible
88  ! We wont overflow, but our strings may get truncated
89  CHARACTER(LEN=51) :: arg

91 
92  num_args = command_argument_count()
93  IF(num_args > 0) THEN
94 
95  ! If this is not the first call, all_args may be already allocated
96  ! Deallocate if needed, and allocate to correct size
97  IF(ALLOCATED(all_args)) DEALLOCATE(all_args)
98  ALLOCATE(all_args(num_args))
99 
100  ! Loop over all arguments
101  DO i = 1, num_args
102  CALL get_command_argument(i, arg)
103  ! Location of the '=' sign
104  ! If not found, return value is 0
105  indx = index(arg, '=')
106  IF(indx > 1) THEN
107  ! All characters up to '=', not including it
108  ! but with any leading spaces removed
109  all_args(i)%name = adjustl(arg(1:indx-1))
110  ! All characters after '='
111  all_args(i)%value= adjustl(arg(indx+1:))
112  ELSE
113  all_args(i)%name = trim(adjustl(arg))
114  ! Value already has a default value, so leave it alone
115  END IF
116  END DO
117  ENDIF
118
