164  ! This function calculates the change in energy if the spin of the randomly
165  ! selected cell were flipped by looking at the spin of neighboring four cells.
166  REAL(real64) function delta_e(grid, n, i, k, j)
167  INTEGER(INT32), INTENT(IN) :: n, i, k
168  INTEGER(INT32), DIMENSION(1:N, 1:N), INTENT(IN) :: grid
169  REAL(real64) :: j
170  delta_e = j * real(grid(i, k) * (grid(i-1, k) + grid(i+1, k) + grid(i, k-1) &
171  + grid(i, k+1)), kind=real64)
172  END FUNCTION delta_e

174  ! This simulates the Ising model for one timestep. Two random numbers between

177  SUBROUTINE simulate(grid, N, beta, J)
178  INTEGER(INT32) :: N
179  INTEGER(INT32), DIMENSION(1:N, 1:N) :: grid
180  INTEGER(INT32) :: rand_cell_i, rand_cell_j
181  REAL(REAL64) :: beta, J, deltaE, P
182 
183  ! Add 2 to each index as we don't want to select cells from the boundary
184  rand_cell_i = int(random()*(n-2), kind=int32) + 2
185  rand_cell_j = int(random()*(n-2), kind=int32) + 2
186 
187  ! We obtain the change in energy of the selected cell's spin were flipped
188  ! and flip the spin as given by the Ising Spin Model algorithm
189  deltae = delta_e(grid, n, rand_cell_i, rand_cell_j, j)
190  IF (deltae < 0) THEN
191  grid(rand_cell_i, rand_cell_j) = -grid(rand_cell_i, rand_cell_j)
192  ELSE IF (deltae > 0) THEN
193  p = exp(-beta*deltae)
194  IF(p > random()) THEN
195  grid(rand_cell_i, rand_cell_j) = -grid(rand_cell_i, rand_cell_j)
196  END IF
197  END IF
198  END SUBROUTINE simulate
