1 ! Written by Steven Tseng
2 ! This program simulates the Ising Spin Model and writes the results to an output
3 ! file 'ising.nc' in the netcdf format.
4 
5 ! An accompanying bash script 'ising.sh' links the necessary libraries, compiles
6 ! the program files and runs the simulation at the values given below in the
7 ! example terminal command. For reproducibility, the initial grid along with the
8 ! command line inputs are written to the output file; these may be utilized as
9 ! initial conditions.
10 
11 ! Example terminal command with arguments for running the program:
12 ! ./ising N=50 T=1000000 init=R beta=2 J=1
13 ! In the above example, the Ferromagnet starts from an ordered state
14 ! and transitions into an unordered state. As shown, the user will enter
15 ! their desired square grid size (N), the number of timesteps to
16 ! iterate (T), the initial state (init), the temperature (beta) and
17 ! the spin-spin interaction strength (J).
18 
19 ! Citations: For the two subroutines that initialize for alternating and random grids,
20 ! I incorporated codes from the model solutions in the DO loops that determine
21 ! if a cell should be set to -1. For writing the data, I used the write_netcdf_array.f90
22 ! module and added additional codes to write all the axes/data to file. Also, as before,
23 ! I am using the modules command_line and random_mod to access command line arguments
24 ! and get random numbers.
25 
26 PROGRAM main
27 
28  ! We import the modules that are utilized by this program.
29  USE sweetener
30  USE write_netcdf
31  USE iso_fortran_env
32 
33  IMPLICIT NONE
34 
35  ! We declare the variables used in the program.
36  ! The first five variables are input through command line arguments.
37  ! N is the size of the NxN grid
38  ! T is the number of timesteps
39  ! init is one of {a or A: Alternating; f or F: Ferromagnet; r or R: Random}
40  ! beta is the temperature
41  ! J is the spin-spin interaction strength
42  INTEGER(INT32) :: n, t
43  CHARACTER(LEN=5) :: init
44  REAL(real64) :: beta, j
45  ! Here we declare the variables that will be stored on the datafile:
46  ! two allocatable rank two arrays for the initial grid and final grid, and
47  ! one allocatable rank one array for the net magnetization at each timestep
48  INTEGER(INT32), DIMENSION(:, :), ALLOCATABLE :: grid, grid_init
49  REAL(real64), DIMENSION(:), ALLOCATABLE :: mag
50  ! Three rank one arrays are declared to be used as the axes for x, y, t
51  INTEGER(INT32), DIMENSION(:), ALLOCATABLE :: xaxis, yaxis, taxis
52  ! ierr variable here is passed to write_data subroutine to store error codes
53  INTEGER(INT32) :: ierr, count = 0
54  ! The data file that will be output at the end is named here
55  CHARACTER(LEN=30) :: filename = "ising.nc"
56 
57  ! We use the subroutine 'parse' to check our command line arguments to make
58  ! sure they are suitable for this simulation
59  CALL parse(n, t, init, beta, j)
60 
61  ! Initialize the grid with the selected/default state whilst storing a copy of
62  ! the initial grid for comparison purposes
63  CALL initialize(grid, grid_init, n, init)
64 
65  ! We allocate the net magnetization array with dimension one greater than the
66  ! total number of timesteps and then set the first value to the net magnetization
67  ! of the initial grid
68  ALLOCATE(mag(t+1))
69  mag(1) = magnetization(grid_init, n)
70 
71  ! Iterate the Ising Spin Model for T timesteps
72  DO WHILE(count < t)
73  CALL simulate(grid, n, beta, j)
74  mag(count+2) = magnetization(grid, n)
75  count = count + 1
76  END DO
77 
78  ! The axes for x, y, t are created here
79  CALL define_axis(xaxis, n, 1)
80  CALL define_axis(yaxis, n, 1)
81  CALL define_axis(taxis, t, 0)
82 
83  ! The data/axes are written to file here
84  CALL write_data(grid, grid_init, mag, xaxis, yaxis, taxis, init, beta, j, &
85  filename, ierr)
86 
87  ! We deallocate allocatble array variables
88  DEALLOCATE(grid, grid_init)
89  DEALLOCATE(mag, xaxis, yaxis, taxis)
90 
91 END PROGRAM main
